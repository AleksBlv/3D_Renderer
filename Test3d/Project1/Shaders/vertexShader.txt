#version 400 core

in vec3 position;
in vec2 texCoord;
in vec3 normal;

uniform mat4 transformation_matrix;
uniform mat4 projection_matrix;
uniform mat4 view_matrix;

out vec2 v_texCoord;

void main(void)
{
    gl_Position = projection_matrix * view_matrix * transformation_matrix * vec4(position, 1.0f); 
    v_texCoord = texCoord; 
}
